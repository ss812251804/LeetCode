给定一个数组 nums，编写一个函数将所有 0 移动到数组的末尾，同时保持非零元素的相对顺序。
示例:
输入: [0,1,0,3,12]
输出: [1,3,12,0,0]
说明:
必须在原数组上操作，不能拷贝额外的数组。
尽量减少操作次数。


两种思路：
1从前向后遍历，遍历到非0时将其前置，遍历完补0
public void moveZeroes(int[] nums) {
             if (nums == null || nums.length == 0) {
                    return;
             } // 记录非o元素开始位置
             int k = 0;
             for (int i = 0; i < nums.length; i++) {
                    if (nums[i] != 0) {
                           nums[k++] = nums[i];
                    }
             }
             while (k < nums.length) {
                    nums[k] = 0;
                    k++;
             }
       }

2.从前向后遍历，遍历到0时将其后置，0后面的往前一位（优化从后往前遍历）
class Solution { 
    public void moveZeroes(int[] nums) { 
        if(nums == null || nums.length == 0) { return; } 
        int j = nums.length; 
        for(int i=0;i<j;){ 
            if(nums[i] == 0) { //(i,j)之间的数字前移 
                for(int k=i+1;k<j;k++){ nums[k-1] = nums[k]; } 
                //移至末尾 nums[j-1] = 0; j--; continue; } i++; 
} } }


class Solution {
     public void moveZeroes(int[] nums) {
         if(nums == null || nums.length == 0) { return; } //定义j表示后面非0的索引 
        int j = nums.length - 1; 
        for(int i=nums.length-1;i>=0;){
             if(nums[i] == 0){ //需要换位置 
                if(i < j){ 
                    for(int k=i+1;k<=j;k++){ 
                        nums[k-1] = nums[k]; } 
                nums[j] = 0; } j--; } i--; } } }
